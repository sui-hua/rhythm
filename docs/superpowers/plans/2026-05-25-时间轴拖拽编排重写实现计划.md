# 时间轴拖拽编排重写 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 重写 Day 页面时间轴任务拖拽编排功能，提供桌面端鼠标实时跟手、5 分钟吸附、`mouseup` 单次保存与失败回弹。

**Architecture:** 以 `useTimelineDragSession` 作为唯一拖拽会话状态入口，统一管理 drag/resize 两类会话、draft 几何状态、吸附与回弹。`TaskItemWrapper.vue` 只保留事件接线与样式组装，`TaskItem.vue` 只负责卡片内容渲染，`useDragSnap.js` 与 `taskLayoutStyle.js` 保持纯计算职责。

**Tech Stack:** Vue 3 Composition API, Vitest, 原生 `mousedown` / `mousemove` / `mouseup`

    session.updateFromMouse({ clientY: 190 })

    expect(session.draftStartHour.value).toBe(null)
    expect(session.draftDurationHours.value).toBe(null)
    expect(session.isActive.value).toBe(false)

### Task 2: 实现 `useTimelineDragSession` 最小闭环

**Files:**

- [ ] **Step 1: 新建 `useTimelineDragSession.js`，定义会话状态与接口**

创建 `src/views/day/composables/useTimelineDragSession.js`，先写最小闭环代码：

```js
import { computed, ref } from 'vue'
import { calcDragResult, calcResizeResult } from '@/views/day/composables/useDragSnap'

export function useTimelineDragSession({ task, onCommit }) {
  const mode = ref('idle')
  const startClientY = ref(0)
  const baseStartHour = ref(task.startHour)
  const baseDurationHours = ref(task.durationHours || 1)
  const draftStartHour = ref(null)
  const draftDurationHours = ref(null)
  const isSaving = ref(false)
  const lastError = ref(null)

  const isActive = computed(() => mode.value !== 'idle')
  const isDragging = computed(() => mode.value === 'drag')
  const isResizing = computed(() => mode.value === 'resize')
  const tooltipHour = computed(() => draftStartHour.value ?? baseStartHour.value)
  const tooltipDuration = computed(() => draftDurationHours.value ?? baseDurationHours.value)

  function startBase(clientY, nextMode) {
    mode.value = nextMode
    startClientY.value = clientY
    baseStartHour.value = task.startHour
    baseDurationHours.value = task.durationHours || 1
    draftStartHour.value = task.startHour
    draftDurationHours.value = task.durationHours || 1
    lastError.value = null
  }

  function startDrag(event) {
    startBase(event.clientY, 'drag')
  }

  function startResize(event) {
    startBase(event.clientY, 'resize')
  }

  function updateFromMouse(event) {
    if (!isActive.value) return
    const deltaY = event.clientY - startClientY.value
    const baseTask = {
      startHour: baseStartHour.value,
      durationHours: baseDurationHours.value
    }

    if (mode.value === 'drag') {
      const result = calcDragResult(baseTask, deltaY)
      draftStartHour.value = result.newStart
      draftDurationHours.value = baseDurationHours.value
      return
    }

    if (mode.value === 'resize') {
      const nextHeight = baseDurationHours.value * 180 + deltaY
      const result = calcResizeResult(baseTask, nextHeight)
      draftStartHour.value = result.newStart
      draftDurationHours.value = result.newEnd - result.newStart
    }
  }

  async function finish() {
    if (!isActive.value || isSaving.value) return

    const nextStart = draftStartHour.value ?? baseStartHour.value
    const nextDuration = draftDurationHours.value ?? baseDurationHours.value
    const nextEnd = nextStart + nextDuration

    if (nextStart === baseStartHour.value && nextDuration === baseDurationHours.value) {
      cancel()
      return
    }

    isSaving.value = true
    try {
      await onCommit({
        newStartHour: nextStart,
        newEndHour: nextEnd
      })
      cancel()
    } catch (error) {
      lastError.value = error
      cancel()
    } finally {
      isSaving.value = false
    }
  }

  function cancel() {
    mode.value = 'idle'
    draftStartHour.value = null
    draftDurationHours.value = null
  }

  return {
    isActive,
    isDragging,
    isResizing,
    isSaving,
    draftStartHour,
    draftDurationHours,
    tooltipHour,
    tooltipDuration,
    startDrag,
    startResize,
    updateFromMouse,
    finish,
    cancel,
    lastError
  }
}
```

- [ ] **Step 1: 重写 `TaskItemWrapper.vue`，删除组件内拖拽公式**

把 `src/views/day/components/TaskItemWrapper.vue` 改成只负责事件接线和样式组装，目标结构如下：

```vue
<script setup>
import { computed, onBeforeUnmount } from 'vue'
import TaskItem from '@/views/day/components/TaskItem.vue'
import DragTimeTooltip from '@/views/day/components/DragTimeTooltip.vue'
import { hourToPx } from '@/views/day/composables/useDragSnap'
import { useTimelineDragSession } from '@/views/day/composables/useTimelineDragSession'
import { useDayData } from '@/views/day/composables/useDayData'
import { buildTaskHorizontalLayoutStyle } from '@/views/day/utils/taskLayoutStyle'

const props = defineProps({
  task: { type: Object, required: true },
  index: { type: Number, required: true }
})

const emit = defineEmits(['select', 'edit'])
const { updateTaskTime } = useDayData()
const canDrag = computed(() => !props.task.completed && props.task.type === 'task')

const session = useTimelineDragSession({
  task: props.task,
  onCommit: ({ newStartHour, newEndHour }) => updateTaskTime(props.task, newStartHour, newEndHour)
})

const displayStartHour = computed(() => session.draftStartHour.value ?? props.task.startHour)
const displayDurationHours = computed(() => session.draftDurationHours.value ?? (props.task.durationHours || 1))

const wrapperStyle = computed(() => ({
  top: `${hourToPx(displayStartHour.value)}px`,
  height: `${hourToPx(displayDurationHours.value)}px`,
  ...buildTaskHorizontalLayoutStyle(props.task)
}))

function handleMouseMove(event) {
  session.updateFromMouse(event)
}

function handleMouseUp() {
  document.removeEventListener('mousemove', handleMouseMove)
  document.removeEventListener('mouseup', handleMouseUp)
  session.finish()
}

function bindDocumentSession() {
  document.addEventListener('mousemove', handleMouseMove)
  document.addEventListener('mouseup', handleMouseUp)
}

function onDragStart(event) {
  session.startDrag(event)
  bindDocumentSession()
}

function onResizeStart(event) {
  session.startResize(event)
  bindDocumentSession()
}

onBeforeUnmount(() => {
  document.removeEventListener('mousemove', handleMouseMove)
  document.removeEventListener('mouseup', handleMouseUp)
  session.cancel()
})
</script>
```
