```js
// src/stores/onboardingStore.js
import { defineStore } from 'pinia'
import { ref } from 'vue'

const defaultSteps = [
  'open-day',
  'create-first-habit',
  'create-first-goal',
  'write-first-summary'
]

export const useOnboardingStore = defineStore('onboarding', () => {
  const visible = ref(false)
  const completedSteps = ref([])
  const activeStep = ref(defaultSteps[0])

  function start() {
    visible.value = true
    activeStep.value = defaultSteps.find((step) => !completedSteps.value.includes(step)) || defaultSteps[0]
  }

  function completeStep(step) {
    if (!completedSteps.value.includes(step)) completedSteps.value.push(step)
    activeStep.value = defaultSteps.find((item) => !completedSteps.value.includes(item)) || null
  }

  function finish() {
    visible.value = false
    activeStep.value = null
  }

  return { visible, completedSteps, activeStep, start, completeStep, finish }
}, {
  persist: {
    pick: ['completedSteps']
  }
})
```

```js
// src/composables/useNotifications.js
import { ref, onUnmounted } from 'vue'
import { playSuccessSound } from '@/utils/audio'

const notifiedTaskIds = ref(new Set())
let checkInterval = null
let swRegistration = null

export function useNotifications() {
  const hasAskedPermission = ref(false)
  const notificationPermission = ref(
    typeof Notification === 'undefined' ? 'unsupported' : Notification.permission
  )

  const requestPermission = async () => {
    hasAskedPermission.value = true
    if (!('Notification' in window)) return false

    if (Notification.permission === 'granted') {
      notificationPermission.value = 'granted'
      swRegistration = await registerServiceWorker()
      return true
    }

    if (Notification.permission === 'denied') return false

    const permission = await Notification.requestPermission()
    notificationPermission.value = permission
    if (permission === 'granted') {
      swRegistration = await registerServiceWorker()
    }
    return permission === 'granted'
  }

  const getPermissionStatus = () => {
    if (!('Notification' in window)) return 'unsupported'
    return Notification.permission
  }

  const showNotification = (title, options = {}) => {
    if (notificationPermission.value !== 'granted') return

    const notification = new Notification(title, {
      icon: '/favicon.ico',
      badge: '/favicon.ico',
      tag: options.tag || 'task-notification',
      ...options
    })

    notification.onclick = () => {
      window.focus()
      notification.close()
    }
  }

  const startListening = (getScheduleItems) => {
    stopListening()

    checkInterval = setInterval(() => {
      if (notificationPermission.value !== 'granted') return

      const items = getScheduleItems()
      if (!items || items.length === 0) return

      items.forEach((item) => {
        if (!item.startHour || notifiedTaskIds.value.has(item.id)) return
        const title = item.type === 'habit' ? `习惯打卡: ${item.title}` : `任务开始: ${item.title}`
        showNotification(title, { tag: `task-${item.id}` })
        notifiedTaskIds.value.add(item.id)
        playSuccessSound()
      })
    }, 30000)
  }

  const stopListening = () => {
    if (!checkInterval) return
    clearInterval(checkInterval)
    checkInterval = null
  }

  const clearNotifiedHistory = () => {
    notifiedTaskIds.value.clear()
  }

  onUnmounted(() => {
    stopListening()
  })

  return {
    notificationPermission,
    hasAskedPermission,
    requestPermission,
    showNotification,
    startListening,
    stopListening,
    clearNotifiedHistory,
    getPermissionStatus
  }
}
```

```js
// src/views/habits/composables/useHabitData.js
export function buildPatchedHabit(habit, newLog, viewContext) {
  const logs = [newLog, ...(habit.logs || [])]
  const monthlyLogs = logs.filter((log) => {
    const date = new Date(log.completed_at)
    return date.getFullYear() === viewContext.year && date.getMonth() === viewContext.month
  })

  return {
    ...habit,
    logs,
    monthlyLogs,
    completedDays: monthlyLogs.map((log) => new Date(log.completed_at).getDate()),
    total: logs.length
  }
}
```

```js
// src/stores/pageStateStore.js
import { defineStore } from 'pinia'
import { reactive } from 'vue'

const buildInitialState = () => ({

export const usePageStateStore = defineStore('pageState', () => {
```

```js
// src/views/day/composables/useDayModal.js
import { computed } from 'vue'
import { usePageStateStore } from '@/stores/pageStateStore'
import { useDayData } from './useDayData'

export function useDayModal() {
  const pageState = usePageStateStore()
  const { dailySchedule } = useDayData()

  const editingTask = computed(() => {
    const index = pageState.day.editingTaskIndex
    if (index === null || !dailySchedule.value) return null
    return dailySchedule.value[index]
  })

  return {
    showAddModal: computed(() => pageState.day.showAddDrawer),
    editingTask,
    openAddModal: () => {
      pageState.day.editingTaskIndex = null
      pageState.day.showAddDrawer = true
    }
  }
}
```
