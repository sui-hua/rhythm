// 修改 useHabitData.js
import { ref, computed } from 'vue'
import { db } from '@/services/database'
import { useDateStore } from '@/stores/dateStore'
import { useHabitStore } from '@/stores/habitStore'

export function useHabitData() {
  const dateStore = useDateStore()
  const habitStore = useHabitStore()

  const isPageLoading = ref(false)
  const selectedHabit = ref(null)
  const currentHabitLogs = ref([])

  const viewYear = ref(dateStore.currentDate.getFullYear())
  const viewMonth = ref(dateStore.currentDate.getMonth())

  // 从 habitStore 读取数据
  const habits = computed(() => habitStore.habits)
  const archivedHabits = computed(() => habitStore.archivedHabits)

  const handleMonthChange = ({ year, month }) => {
    viewYear.value = year
    viewMonth.value = month
    if (selectedHabit.value && currentHabitLogs.value.length > 0) {
      const monthlyLogs = currentHabitLogs.value.filter((log) => {
        const d = new Date(log.completed_at)
        return d.getFullYear() === year && d.getMonth() === month
      })
      const completedDays = monthlyLogs.map((log) => new Date(log.completed_at).getDate())
      selectedHabit.value = {
        ...selectedHabit.value,
        monthlyLogs,
        completedDays
      }
    }
  }

  const calculateStreak = (logs) => {
    if (!logs || logs.length === 0) return 0
    const sortedDates = [...new Set(
      logs.map(log => new Date(log.completed_at).toDateString())
    )].sort().reverse()
    if (sortedDates.length === 0) return 0
    let streak = 0
    const today = new Date().toDateString()
    const yesterday = new Date(Date.now() - 86400000).toDateString()
    const startDate = sortedDates[0] === today ? today :
                      sortedDates[0] === yesterday ? yesterday : null
    if (!startDate) return 0
    let currentDate = new Date(startDate)
    for (const dateStr of sortedDates) {
      const expected = currentDate.toDateString()
      if (dateStr === expected) {
        streak++
        currentDate = new Date(currentDate.getTime() - 86400000)
      } else if (new Date(dateStr) < currentDate) {
        break
      }
    }
    return streak
  }

  const fetchHabits = async () => {
    isPageLoading.value = true
    try {
      await habitStore.fetchHabits()
      if (selectedHabit.value) {
        const updated = habitStore.allHabits.find((h) => h.id === selectedHabit.value.id)
        if (updated) selectedHabit.value = updated
        else selectedHabit.value = null
      } else if (habits.value.length > 0) {
        selectedHabit.value = habits.value[0]
      } else if (archivedHabits.value.length > 0) {
        selectedHabit.value = archivedHabits.value[0]
      }
    } catch (e) {
      console.error('Fetch habits failed', e)
    } finally {
      isPageLoading.value = false
    }
  }

  const fetchLogsForHabit = async (habitId) => {
    if (!habitId) {
      currentHabitLogs.value = []
      return
    }
    try {
      const logs = await db.habit.listLogsByHabit(habitId)
      currentHabitLogs.value = logs
      const habit = habitStore.allHabits.find(h => h.id === habitId)
      if (habit) {
        const viewYearVal = viewYear.value
        const viewMonthVal = viewMonth.value
        const monthlyLogs = logs.filter((log) => {
          const d = new Date(log.completed_at)
          return d.getFullYear() === viewYearVal && d.getMonth() === viewMonthVal
        })
        const completedDays = monthlyLogs.map((log) => new Date(log.completed_at).getDate())
        const updatedHabit = {
          ...habit,
          logs,
          monthlyLogs,
          completedDays,
          total: logs.length,
          completionRate: Math.round((logs.length / 30) * 100),
          streak: calculateStreak(logs)
        }
        habitStore.patchHabit(habitId, updatedHabit)
      }
    } catch (e) {
      console.error('Fetch logs for habit failed', e)
    }
  }

  return {
    habits,
    archivedHabits,
    selectedHabit,
    viewYear,
    viewMonth,
    handleMonthChange,
    fetchHabits,
    fetchLogsForHabit,
    isPageLoading
  }
}

export function buildPatchedHabit(habit, newLog, viewContext) {
  if (!newLog) return habit
  const logs = [newLog, ...(habit.logs || [])]
  const monthlyLogs = logs.filter((log) => {
    const date = new Date(log.completed_at)
    return date.getFullYear() === viewContext.year && date.getMonth() === viewContext.month
  })
  return {
    ...habit,
    logs,
    monthlyLogs,
    completedDays: monthlyLogs.map((log) => new Date(log.completed_at).getDate()),
    total: logs.length
  }
}

            return d.getFullYear() === viewYear.value && d.getMonth() === viewMonth.value
          }),
          completedDays: updatedLogs.filter((log) => {
            const d = new Date(log.completed_at)
            return d.getFullYear() === viewYear.value && d.getMonth() === viewMonth.value
          }).map((log) => new Date(log.completed_at).getDate()),
          total: updatedLogs.length
        })
      } else {
        const date = new Date(viewYear.value, viewMonth.value, day, 12, 0, 0)
        const newLog = await db.habit.log(habit.id, '', date.toISOString())
        // 乐观更新：添加新日志
        const patched = buildPatchedHabit(habit, newLog, { year: viewYear.value, month: viewMonth.value })
        habitStore.patchHabit(habit.id, {
          logs: patched.logs,
          monthlyLogs: patched.monthlyLogs,
          completedDays: patched.completedDays,
          total: patched.total
        })
      }
    } catch (e) {
      console.error('Toggle habit log failed', e)
      await fetchHabits()
    } finally {
      isSubmitting.value = false
    }
  }

  const handleQuickLog = async (note) => {
    if (!selectedHabit.value || !note.trim() || isSubmitting.value) {
      return false
    }
    isSubmitting.value = true
    try {
      const now = new Date()
      const today = now.getDate()
      const existingLog = selectedHabit.value.monthlyLogs.find((log) => {
        return new Date(log.completed_at).getDate() === today
      })
      if (!existingLog) {
        const date = new Date(now.getFullYear(), now.getMonth(), today, 12, 0, 0)
        const newLog = await db.habit.log(selectedHabit.value.id, note.trim(), date.toISOString())
        // 乐观更新
        const habit = selectedHabit.value
        const patched = buildPatchedHabit(habit, newLog, { year: viewYear.value, month: viewMonth.value })
        habitStore.patchHabit(habit.id, {
          logs: patched.logs,
          monthlyLogs: patched.monthlyLogs,
          completedDays: patched.completedDays,
          total: patched.total
        })
      }
      return true
    } catch (e) {
      console.error('Quick log failed', e)
      return false
    } finally {
      isSubmitting.value = false
    }
  }

  return {
    toggleComplete,
    handleQuickLog,
    isSubmitting
  }
}

// src/stores/goalDataStore.js
import { defineStore } from 'pinia'
import { ref, reactive } from 'vue'

export const useGoalDataStore = defineStore('goalData', () => {
  const goals = ref([])
  const goalMonths = ref([])
  const goalMonthsCache = reactive({})
  const goalDaysCache = reactive({})
  const archiveVersion = ref(0)
  const categories = ref([])
  const initialized = ref(false)

  const getGoalMonthsByGoalId = (goalId) => goalMonthsCache[goalId] || []

  const clearGoalDaysCache = (keepMonthPlanId = null) => {
    if (keepMonthPlanId && goalDaysCache[keepMonthPlanId]) {
      const keepData = goalDaysCache[keepMonthPlanId]
      Object.keys(goalDaysCache).forEach(key => {
        delete goalDaysCache[key]
      })
      goalDaysCache[keepMonthPlanId] = keepData
    } else {
      Object.keys(goalDaysCache).forEach(key => {
        delete goalDaysCache[key]
      })
    }
  }

  const syncGoalMonthsToFlatList = (goalId) => {
    const cached = goalMonthsCache[goalId] || []
    const others = goalMonths.value.filter(item => item.goal_id !== goalId)
    goalMonths.value = [...others, ...cached]
  }

  const reset = () => {
    goals.value = []
    goalMonths.value = []
    initialized.value = false
    categories.value = []
    archiveVersion.value = 0
    Object.keys(goalMonthsCache).forEach(key => delete goalMonthsCache[key])
    Object.keys(goalDaysCache).forEach(key => delete goalDaysCache[key])
  }

  return {
    goals, goalMonths, goalMonthsCache, goalDaysCache, archiveVersion,
    categories, initialized,
    getGoalMonthsByGoalId, clearGoalDaysCache, syncGoalMonthsToFlatList, reset
  }
})

          :style="{

            top: 0,
            left: 0,
            width: '100%',
            transform: `translateY(${item.start}px)`
          }"

            :task="item.data"

import { ref, computed } from 'vue'
import { useVirtualizer } from '@tanstack/vue-virtual'
import { ScrollArea } from '@/components/ui/scroll-area'

import TimelineMarker from '@/views/day/components/TimelineMarker.vue'
import { useDayStore } from '@/stores/dayStore'
import { buildTimelineDisplaySchedule } from '@/views/day/utils/timelineLayout'

const props = defineProps({

  count: displaySchedule.value.length,
  getScrollElement: () => scrollContainerRef.value?.viewportElement,
  estimateSize: () => ITEM_HEIGHT,
  overscan: 5

const virtualItems = computed(() => virtualizer.value.getVirtualItems())
const totalHeight = computed(() => virtualizer.value.getTotalSize())

  return Array.from({ length: endHour - startHour }, (_, i) => startHour + i + 1)
})

defineExpose({
  timelineContainer: computed(() => timelineContainerRef.value?.viewportElement)
})

defineEmits(['edit-task', 'select-task'])
