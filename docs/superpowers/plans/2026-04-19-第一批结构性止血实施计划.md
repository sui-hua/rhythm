# 第一批结构性止血批 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use `subagent-driven-development` or `executing-plans` to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 完成第一批结构性止血改造：收敛为单一 `db`、稳定 `Direction` 请求与缓存、修复 `Habits/Direction` 移动端结构性不可用、落地时序页双层结构最小版，并统一高频写操作反馈。

**Architecture:** 这一批按“垂直切片”执行，而不是先做完某一层再做下一层。先统一数据访问契约与 `Direction` 状态边界，再拆 `Habits/Direction` 的端特定页面壳，最后在稳定的数据模型上改造时序页的统一执行项视图、移动端任务流与反馈体系。所有页面继续共享同一套业务数据层与 composable，不允许端特定页面壳复制业务逻辑。

**Tech Stack:** Vue 3 Composition API、Vite、Pinia、Vue Router、Vitest、Supabase

```js
// src/stores/directionStore.js
import { defineStore } from 'pinia'
import { ref, reactive } from 'vue'

export const useDirectionStore = defineStore('direction', () => {
  const plans = ref([])
  const monthlyPlans = ref([])
  const selectedGoal = ref(null)
  const selectedMonth = ref(null)
  const monthlyPlansCache = reactive({})
  const dailyPlansCache = reactive({})
  const initialized = ref(false)

  function reset() {
    plans.value = []
    monthlyPlans.value = []
    selectedGoal.value = null
    selectedMonth.value = null
    Object.keys(monthlyPlansCache).forEach((key) => delete monthlyPlansCache[key])
    Object.keys(dailyPlansCache).forEach((key) => delete dailyPlansCache[key])
    initialized.value = false
  }

  return { plans, monthlyPlans, selectedGoal, selectedMonth, monthlyPlansCache, dailyPlansCache, initialized, reset }
})
```

```js
// src/views/direction/composables/useDirectionFetch.js
const inflightDaily = new Map()

async function ensureDailyPlans(monthlyPlanId) {
  if (inflightDaily.has(monthlyPlanId)) return inflightDaily.get(monthlyPlanId)

  const promise = db.dailyPlans.list(monthlyPlanId).finally(() => {
    inflightDaily.delete(monthlyPlanId)
  })

  inflightDaily.set(monthlyPlanId, promise)
  const rows = await promise
  store.dailyPlansCache[monthlyPlanId] = rows
  return rows
}

async function selectMonthForGoal(planId, month) {
  store.selectedMonth = month
  const mp = (store.monthlyPlansCache[planId] || []).find((item) => getDateOnlyMonth(item.month) === month)
  if (mp) await ensureDailyPlans(mp.id)
}
```

```js
// src/views/day/composables/useDayExecutionItems.js
export function buildDayExecutionItems({ tasks, dailyPlans, habits, habitLogs }) {
  const habitLogIds = new Set(habitLogs.map((log) => log.habit_id))

  return [
    ...habits.map((habit) => ({
      id: habit.id,
      type: 'habit',
      title: habit.title,
      time: habit.task_time || '未安排',
      startHour: habit.task_time ? toStartHour(habit.task_time) : undefined,
      completed: habitLogIds.has(habit.id),
      sourceLabel: '习惯',
      sourceMeta: {},
      original: habit
    })),
    ...tasks.map((task) => ({
      id: task.id,
      type: 'task',
      title: task.title,
      time: toTimeText(task.start_time),
      startHour: toStartHour(task.start_time),
      completed: task.completed,
      sourceLabel: '时序',
      sourceMeta: {},
      original: task
    })),
    ...dailyPlans.map((plan) => ({
      id: plan.id,
      type: 'daily_plan',
      title: plan.title,
      time: plan.task_time || plan.monthly_plans?.task_time || plan.monthly_plans?.plans?.task_time || '未安排',
      startHour: toStartHour(plan.task_time || plan.monthly_plans?.task_time || plan.monthly_plans?.plans?.task_time),
      completed: isDailyPlanCompleted(plan.status),
      sourceLabel: '所向',
      sourceMeta: {
        planId: plan.monthly_plans?.plans?.id,
        planTitle: plan.monthly_plans?.plans?.title || ''
      },
      original: plan
    }))
  ].sort(sortBySchedule)
}
```
