```sql

ALTER TABLE goal_days ADD CONSTRAINT goal_days_goal_month_id_fkey
  FOREIGN KEY (goal_month_id) REFERENCES goal_months(id) ON DELETE CASCADE;

ALTER TABLE goal DROP CONSTRAINT goal_category_id_fkey;
ALTER TABLE goal ADD CONSTRAINT goal_category_id_fkey
  FOREIGN KEY (category_id) REFERENCES goal_categories(id) ON DELETE SET NULL;
```

- [ ] **Step 3: 执行 status 类型统一**

```sql
ALTER TABLE goal_days ADD COLUMN status_new varchar DEFAULT 'active';
UPDATE goal_days SET status_new = CASE WHEN status = 0 THEN 'active' WHEN status = 1 THEN 'completed' ELSE 'active' END;
ALTER TABLE goal_days DROP COLUMN status;
ALTER TABLE goal_days RENAME COLUMN status_new TO status;
ALTER TABLE goal_days ADD CONSTRAINT goal_days_status_check
  CHECK (status IN ('active', 'completed', 'archived'));
```

- [ ] **Step 4: 执行 updated_at 触发器**

```sql
CREATE OR REPLACE FUNCTION update_updated_at()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = now();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER set_updated_at BEFORE UPDATE ON task FOR EACH ROW EXECUTE FUNCTION update_updated_at();
CREATE TRIGGER set_updated_at BEFORE UPDATE ON goal FOR EACH ROW EXECUTE FUNCTION update_updated_at();
CREATE TRIGGER set_updated_at BEFORE UPDATE ON goal_months FOR EACH ROW EXECUTE FUNCTION update_updated_at();
CREATE TRIGGER set_updated_at BEFORE UPDATE ON goal_days FOR EACH ROW EXECUTE FUNCTION update_updated_at();
CREATE TRIGGER set_updated_at BEFORE UPDATE ON habit FOR EACH ROW EXECUTE FUNCTION update_updated_at();
CREATE TRIGGER set_updated_at BEFORE UPDATE ON habit_logs FOR EACH ROW EXECUTE FUNCTION update_updated_at();
CREATE TRIGGER set_updated_at BEFORE UPDATE ON summary FOR EACH ROW EXECUTE FUNCTION update_updated_at();
```

- [ ] **Step 5: 执行索引优化**


新建 `src/utils/safeAction.ts`：

```typescript
/**
 * 统一的异步操作错误处理包装
 * 捕获错误并输出到 console，未来可接入 toast 通知
 */
export async function safeAction<T>(
  action: () => Promise<T>,
  errorMessage: string
): Promise<T | null> {
  try {
    return await action()
  } catch (e) {
    console.error(errorMessage, e)
    return null
  }
}
```
